# 180313 문제 1

## Question

```
  // 성적 받기 복습
  // 총점과 평균 반환
```
 
## Solution

```javascript
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace ConsoleApp3
{
    class Program
    {
        static void Main(string[] args)
        {
            int[] myCoord = new int[2] { 1, 1 };
            int[] checkPosition = new int[2];

            int character1 = 345;
            int character2 = 346;
            int singleTree = 10;
            int doubleTree = 11;
            int blackStone = 100;
            int redStone = 101;
            int portalAtoB = 999;
            int portalBtoA = 990;
            bool isPlayerSpawned = false;

            int[,] map = new int[11, 11];

            string commandInput = "";

            
            while (true)
            {

                for (int x = 0; x < 11; x++)
            {
                map[x, 0] = 1;
                map[x, 10] = 1;
            }
            for (int y = 0; y < 11; y++)
            {
                map[0, y] = 1;
                map[10, y] = 1;
            }

                for (int x = 1; x < 10; x++)
                {
                    map[x, 1] = 0;
                    map[x, 9] = 0;
                }
                for (int y = 1; y < 10; y++)
                {
                    map[1, y] = 0;
                    map[9, y] = 0;
                }

                map[9, 8] = portalAtoB;

            map[3, 1] = blackStone;
            map[4, 4] = redStone;
            map[5, 7] = character2;
            map[3, 7] = character1;
            for (int i = 2; i < 6; i++)
            {
                map[i, 3] = doubleTree;
            }
            for (int i = 2; i < 8; i++)
            {
                map[i, 6] = doubleTree;
            }
            for (int i = 3; i < 6; i++)
            {
                map[7, i] = doubleTree;
            }
            for (int i = 3; i < 7; i++)
            {
                map[1, i] = singleTree;
            }

                //spawn player
                if (isPlayerSpawned == false)
                {
                    map[myCoord[0], myCoord[1]] = -1;
                    isPlayerSpawned = true;
                }

                map[myCoord[0], myCoord[1]] = -1;
                

                //draw map
                for (int y = 0; y < 11; y++)
                {
                    for (int x = 0; x < 11; x++)
                    {
                        if (map[x, y] < 100)
                        {
                            Console.Write(" ");
                        }
                        if (map[x, y] < 10 && map[x, y] > -1)
                        {
                            Console.Write(" ");
                        }
                        Console.Write(map[x, y] + " ");


                    }
                    Console.WriteLine("");
                }




                Console.WriteLine("이동할 방향을 입력하세요 : 동, 서, 남, 북");
                commandInput = Console.ReadLine();

                switch (commandInput)
                {
                    case ("동"):
                        checkPosition = (int[])myCoord.Clone();
                        checkPosition[0]++;
                        if (map[checkPosition[0], checkPosition[1]] != 0 && map[checkPosition[0], checkPosition[1]] < 900)
                        {
                            Console.WriteLine("막힌 구역입니다.");
                            Console.WriteLine("내 위치 : " + myCoord[0] + ", " + myCoord[1]);
                        }

                        else
                        {
                            myCoord[0]++;
                            Console.WriteLine("동쪽으로 이동했습니다.");
                            Console.WriteLine("내 위치 : " + myCoord[0] + ", " + myCoord[1]);
                        }
                        break;

                    case ("서"):
                        checkPosition = (int[])myCoord.Clone();
                        checkPosition[0]--;
                        if (map[checkPosition[0], checkPosition[1]] != 0 && map[checkPosition[0], checkPosition[1]] < 900)
                        {
                            Console.WriteLine("막힌 구역입니다.");
                            Console.WriteLine("내 위치 : " + myCoord[0] + ", " + myCoord[1]);
                        }

                        else
                        {
                            myCoord[0]--;
                            Console.WriteLine("동쪽으로 이동했습니다.");
                            Console.WriteLine("내 위치 : " + myCoord[0] + ", " + myCoord[1]);
                        }
                        break;
                    case ("남"):
                        checkPosition = (int[])myCoord.Clone();
                        checkPosition[1]++;
                        if (map[checkPosition[0], checkPosition[1]] != 0 && map[checkPosition[0], checkPosition[1]] < 900)
                        {
                            Console.WriteLine("막힌 구역입니다.");
                            Console.WriteLine("내 위치 : " + myCoord[0] + ", " + myCoord[1]);
                        }

                        else
                        {
                            myCoord[1]++;
                            Console.WriteLine("동쪽으로 이동했습니다.");
                            Console.WriteLine("내 위치 : " + myCoord[0] + ", " + myCoord[1]);
                        }
                        break;
                    case ("북"):
                        checkPosition = (int[])myCoord.Clone();
                        checkPosition[1]--;
                        if (map[checkPosition[0], checkPosition[1]] != 0 && map[checkPosition[0], checkPosition[1]] < 900)
                        {
                            Console.WriteLine("막힌 구역입니다.");
                            Console.WriteLine("내 위치 : " + myCoord[0] + ", " + myCoord[1]);
                        }

                        else
                        {
                            myCoord[1]--;
                            Console.WriteLine("동쪽으로 이동했습니다.");
                            Console.WriteLine("내 위치 : " + myCoord[0] + ", " + myCoord[1]);
                        }
                        break;
                }

            }
        }
    }
}
```
