# 180222 과제 1

## Solution

```javascript
public class Test : MonoBehaviour {

	void Start () {
        int number;
        int factor;
        int result;
        string strTempResult = "";
        string strResult = "";

        for (factor = 1; factor < 10; factor++)
        {
            for (number = 1; number < 10; number++)
            {
                result = number * factor;
                strTempResult = number + " x " + factor + " = " + result;

                if (number == 1)
                {
                    strResult = strResult + strTempResult;
                }else
                {
                    strResult = strResult + " , " + strTempResult;
                }
            }
            Debug.Log(strResult);
            strResult = "";
        }
    }

	void Update () {
		
	}
```
