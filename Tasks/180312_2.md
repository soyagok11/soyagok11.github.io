# 180312 문제 2

## Question

```
  // 길바닥에 떨어진 아이템을 줍고 버리는 시스템
```
 
## Solution

```javascript
using System;

namespace ConsoleApp3
{
    class Program
    {
        static void Main(string[] args)
        {
            string commandInput;

            string[] items = new string[3] { "사탕", "못", "동전" };
            string[] bags = new string[items.Length];


            START:

            for(int i = 0; i < items.Length; i++)
            {
                if(items[i] != ""){
                    goto ITEM_GET_ENABLE;
                }
            }
            Console.WriteLine("길바닥에 아이템이 없습니다.");
            goto GET_CONSOLE_COMMAND;


            ITEM_GET_ENABLE:

            Console.Write("길바닥에 아이템 ");
            for (int i = 0; i < items.Length; i++)
            {
                if (items[i] != "")
                {
                    if (i == items.Length - 1)
                    {
                        Console.Write(items[i]);
                    }

                    else
                    {
                        Console.Write(items[i] + ", ");
                    }
                }
                else
                {
                    Console.Write(items[i]);
                }
            }
            
            Console.WriteLine("이 떨어져 있습니다.");



            GET_CONSOLE_COMMAND:

            Console.WriteLine("행동을 입력해주세요.\n (아이템) (줍다) \n (아이템) (버리다)\n");
            commandInput = Console.ReadLine();

            if (commandInput.Contains(" 줍다"))
            {
                if (commandInput.Contains("사탕"))
                {
                    if (items[0] == "사탕")
                    {
                        bags[0] = items[0];
                        items[0] = "";
                        goto GET_ITEM_MESSAGE;
                    }
                    else
                    {
                        goto ERROR_NO_ITEM;
                    }
                }
                else if (commandInput.Contains("못"))
                {
                    if (items[1] == "못")
                    {
                        bags[1] = items[1];
                        items[1] = "";
                        goto GET_ITEM_MESSAGE;
                    }
                    else
                    {
                        goto ERROR_NO_ITEM;
                    }
                }
                else if (commandInput.Contains("동전"))
                {
                    if (items[2] == "동전")
                    {
                        bags[2] = items[2];
                        items[2] = "";
                        goto GET_ITEM_MESSAGE;
                    }
                    else
                    {
                        goto ERROR_NO_ITEM;
                    }
                }
                else
                {
                    goto ERROR_NO_ITEM;
                }
            }
            else if (commandInput.Contains(" 버리다"))
            {
                if (commandInput.Contains("사탕"))
                {
                    if (bags[0] == "사탕")
                    {
                        items[0] = bags[0];
                        bags[0] = "";
                        goto THROWAWAY_ITEM_MESSAGE;
                    }
                    else
                    {
                        goto ERROR_NO_ITEM;
                    }
                }
                else if (commandInput.Contains("못"))
                {
                    if (bags[1] == "못")
                    {
                        items[1] = bags[1];
                        bags[1] = "";
                        goto THROWAWAY_ITEM_MESSAGE;
                    }
                    else
                    {
                        goto ERROR_NO_ITEM;
                    }
                }
                else if (commandInput.Contains("동전"))
                {
                    if (bags[2] == "동전")
                    {
                        items[2] = bags[2];
                        bags[2] = "";
                        goto THROWAWAY_ITEM_MESSAGE;
                    }
                    else
                    {
                        goto ERROR_NO_ITEM;
                    }
                }
                else
                {
                    goto ERROR_NO_ITEM;
                }
            }
            else
            {
                goto ERROR_NO_COMMAND;
            }


            GET_ITEM_MESSAGE:
            Console.WriteLine("아이템을 주웠습니다.");
            goto START;

            THROWAWAY_ITEM_MESSAGE:
            Console.WriteLine("아이템을 버렸습니다.");
            goto START;

            ERROR_NO_ITEM:
            Console.WriteLine("그런 아이템은 존재하지 않습니다.");
            goto START;

            ERROR_NO_COMMAND:
            Console.WriteLine("존재하지 않는 명령어입니다.");
            goto START;

        }
    }
}

```

## Notes

바닥에 있는 특정 아이템을 집으라는줄 알고 질문과는 다른 것을 만들었다.

이쁘게 보이게 하려고 지저분하게 만들었다. 그래도 함수를 쓰기 전에는 goto문을 계속 쓸 듯. 값을 건네줄 수 없어서 불편하지만 그나마 깔끔하게 나온다.
