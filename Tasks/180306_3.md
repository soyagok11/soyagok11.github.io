# 180306 문제 3

## Question

```
			//지갑에 돈을 넣는다. [입력]

            //지갑에는 2000원이 있습니다.

            //"상품 입력"
            //원하는 상품을 입력 하세요. 상품의 이름에 마추어 id를 설정하고 가격을 설정하세요. id, 가격 본인 마음대로 설정 합니다.

            //"장바구니 봐"
            //자갈치, 서울우유, 폴라포.....

            //"전체가격"
            //합계 : 000원 

            //"구매" (장바구니에 물건이 있을 경우)
            //돈이 부족하다면 
            //구매 할수 없습니다. (장바구니를 비워주세요)
            //돈이 부족 하지 않다면 
            // 구매 되었습니다. 지갑에 남은 금액은 : 0000입니다.

            //"구매 목록"
            // 있을 경우 : 자갈치, 서울우유, 폴라포....
            // 없을 경우 : 구매 내역이 없습니다.

            //"구매종료"
            //프로그램을 종료 합니다.

```
 
## Solution

```javascript
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace fruit
{
    class Program
    {
        static void Main(string[] args)
        {
            int myMoney = 0;
            int productPrice = 0;
            bool forLoopExitBoolValue = true;
            string commandText = "";
            string productName = "";
            int productID = 0;
            string basketList = "";
            int productPriceSum = 0;
            string boughtProductList = "";

            Console.Write("당신의 지갑에는 얼마가 있습니까? : ");
            myMoney = int.Parse(Console.ReadLine());
            Console.WriteLine("당신의 지갑에는 " + myMoney + "원이 있습니다.");
            Console.WriteLine("편의점에 도착했습니다. 원하는 행동을 입력해 주세요.");

            for (int i = 0; i < 10; i++)
            {
                Console.WriteLine("* 명령어 : [상품 입력], [장바구니 봐], [장바구니 비워], [전체 가격], [구매], [구매 목록], [구매 종료] *");

                commandText = Console.ReadLine();

                switch (commandText)
                {
                    case "상품 입력":

                        //상품 리스트
                        Console.WriteLine("구매하려는 상품의 번호를 입력해주세요.");
                        for (int j = 1; forLoopExitBoolValue; j++)
                        {
                            switch (j)
                            {
                                case (1):
                                    productName = "코카콜라";
                                    productPrice = 500;
                                    break;
                                case (2):
                                    productName = "펩시콜라";
                                    productPrice = 450;
                                    break;
                                case (3):
                                    productName = "스프라이트";
                                    productPrice = 400;
                                    break;
                                case (4):
                                    productName = "조지아";
                                    productPrice = 700;
                                    break;
                                case (5):
                                    productName = "데미소다";
                                    productPrice = 350;
                                    break;
                                case (6):
                                    productName = "나랑드 사이다";
                                    productPrice = 300;
                                    forLoopExitBoolValue = false;
                                    break;
                                default:
                                    break;
                            }
                            Console.WriteLine("[" + j + "] " + productName + " : " + productPrice + "원");
                        }
                        forLoopExitBoolValue = true;

                        productID = int.Parse(Console.ReadLine());

                        switch (productID)
                        {
                            case (1):
                                productID = 1;
                                productName = "코카콜라";
                                productPrice = 500;
                                break;
                            case (2):
                                productID = 2;
                                productName = "펩시콜라";
                                productPrice = 450;
                                break;
                            case (3):
                                productID = 3;
                                productName = "스프라이트";
                                productPrice = 400;
                                break;
                            case (4):
                                productID = 4;
                                productName = "조지아";
                                productPrice = 700;
                                break;
                            case (5):
                                productID = 5;
                                productName = "데미소다";
                                productPrice = 350;
                                break;
                            case (6):
                                productID = 6;
                                productName = "나랑드 사이다";
                                productPrice = 300;
                                break;
                            default:
                                Console.WriteLine("해당 번호에는 제품이 존재하지 않습니다.");
                                break;
                        }

                        if (productID > 0 && productID < 7)
                        {
                            basketList = basketList + productName + " ";
                            productPriceSum += productPrice;

                            Console.WriteLine(productName + "을 장바구니에 담았습니다.");
                        }

                        i = 0;
                        break;
                    case "장바구니 봐":
                        if(basketList == "")
                        {
                            Console.WriteLine("장바구니에 물건이 없습니다.");
                        }
                        else { 
                        Console.WriteLine("장바구니 : " + basketList);
                        }
                        i = 0;
                        break;
                    case "장바구니 비워":
                        basketList = "";
                        Console.WriteLine("장바구니를 비웠습니다.");
                        break;
                    case "전체 가격":
                        Console.Write("전체 가격 : ");
                        Console.WriteLine(productPriceSum);
                        i = 0;
                        break;
                    case "구매":
                        if(myMoney - productPriceSum < 0)
                        {
                            Console.WriteLine("잔액이 부족합니다.");
                        }
                        else
                        {
                            myMoney -= productPriceSum;
                            boughtProductList += basketList;
                            basketList = "";
                            Console.WriteLine("장바구니의 물건들을 구매했습니다.");
                            Console.WriteLine("지갑에 " + myMoney + "원이 남았습니다.");
                        }
                        i = 0;
                        break;
                    case "구매 목록":
                        if(boughtProductList == "")
                        {
                            Console.WriteLine("구매한 물건이 없습니다.");
                        }
                        else { 
                        Console.Write("내 구매 목록 : ");
                        Console.WriteLine(boughtProductList);
                        }
                        i = 0;
                        break;
                    case "구매 종료":
                        Console.WriteLine("구매를 종료합니다.");
                        Console.ReadKey();
                        i = 11;
                        break;
                    default:
                        Console.WriteLine("잘못된 명령어입니다.");
                        i = 0;
                        break;
                }
            }
        }
    }
}
```
