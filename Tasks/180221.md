## Solution

public class Task : MonoBehaviour {

	void Start () {
        int number=9;
        int factor;
        int product;

        for (factor = 1; factor <= 9; factor++)
        {
            product = number * factor;
            Debug.Log(number + " x " + factor + " = " + product);
        }
		
	}
	
	void Update () {
		
	}
}
